CREATE PROCEDURE [dbo].[BusquedaClienteAlbaran] @NumeroAlbaran nvarchar(30)
AS
select 
A.NumeroAlbaran
, FORMAT(A.Fecha, 'dd/MM/yyyy') as Fecha
, C.NIF
, A.Total
, C.Nombre
, D.LineaDireccion
, D.Poblacion
, D.Provincia
, D.CodigoPostal
from Albaran A
inner join Cliente C on C.Id = A.ClienteId
inner join Direccion D on D.Id = C.DireccionId
Where A.NumeroAlbaran = @NumeroAlbaran
GO

CREATE PROCEDURE [dbo].[BusquedaClienteFactura] @NumeroFactura nvarchar(30)
AS
select F.NumeroFactura
, FORMAT(F.Fecha, 'dd/MM/yyyy') as Fecha, C.NIF, F.Total, C.Nombre, D.LineaDireccion, D.Poblacion, D.Provincia, D.CodigoPostal
from Factura F
inner join Cliente C on C.Id = F.ClienteId
inner join Direccion D on D.Id = C.DireccionId
Where NumeroFactura = @NumeroFactura
GO

CREATE PROCEDURE [dbo].[BusquedaLineasAlbaran] @NumeroAlbaran nvarchar(30)

, LA.Cajas

GO

CREATE PROCEDURE [dbo].[BusquedaLineasFactura] @NumeroFactura nvarchar(30)

, P.ZonaCaptura
, LF.Cajas
, LF.Kgs

GO

CREATE PROCEDURE [dbo].[BusquedaLineasIVA] @NumeroFactura nvarchar(30)
AS
select I.Id as IVAId, I.Porcentaje, I.PorcentanjeRE, SUM(LF.Importe) as BaseIVA, SUM(LF.ImporteIVA) as ImporteIVA, SUM(LF.ImporteRE) as ImporteRE
from LineaFactura LF
inner join Producto P on P.Id = LF.ProductoId
inner join IVA I on I.Id = P.IVAId
where LF.NumeroFactura = @NumeroFactura
group by I.Id, I.Porcentaje, I.PorcentanjeRE 
GO

CREATE PROCEDURE [dbo].[BusquedaLineasIVAAlbaran] @NumeroAlbaran nvarchar(30)
AS
select I.Id as IVAId
, I.Porcentaje
, I.PorcentanjeRE
, SUM(LA.Importe) as BaseIVA
, SUM(LA.ImporteIVA) as ImporteIVA
, SUM(LA.ImporteRE) as ImporteRE
from LineaAlbaran LA
inner join Producto P on P.Id = LA.ProductoId
inner join IVA I on I.Id = P.IVAId
where LA.NumeroAlbaran = @NumeroAlbaran
group by I.Id, I.Porcentaje, I.PorcentanjeRE
